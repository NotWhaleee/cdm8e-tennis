        # movement of right platform
        ldi r0, 0x1e # x of right platform 
        ld r0, r0
        inc r0
        ldi r1, 0x1f # 00011111
        and r0, r1
        ldi r2, yBall
        ld r2, r2
        if
            cmp r2, r1
        is lo
            jsr go_down_pl_right
        else 
            jsr go_up_pl_right

        fi
